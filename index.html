var $searchField = $('#search-field');

// timer for performing instant search
var typingTimer;
var doneTypingInterval = 300; // timeout for new in-page search

//on keyup, start the countdown
$searchField.keyup(function() {
  clearTimeout(typingTimer);
  typingTimer = setTimeout(inPageSearch, doneTypingInterval);
});

//on keydown, clear the countdown
$searchField.keydown(function() {
  clearTimeout(typingTimer);
});

function addHidingClass($div) {
  if (!$div.hasClass('hiding-class')) {
    $div.addClass('hiding-class');
  }
}

function removeHidingClass($div) {
  $div.removeClass('hiding-class');
}

function inPageSearch() {
  var searchQuery = $searchField.val().trim().toLowerCase();

  $('.person').each(function() {
    var testQuery = $(this).text().trim().toLowerCase();
    if (testQuery.match(searchQuery) == null) {
      addHidingClass($(this));
    }
    else {
      removeHidingClass($(this));
    }
  });
}

jQuery AJAX

Asynchronous JavaScript and XML

  var detective = { name: 'Sherlock Holmes' }; // a javascript object

  $.ajax({
      type: 'POST',                            // http method
      url: '/api/addDetective/',               // target URL
      data: detective,                         // data (the js obj)
      success: function(serverResp) {          // on ajax success
          if (serverResp.code == '200') {
              console.log('detective added successfully!');
          }
          else {
              console.log('detective already exists!');
          }
      },
      error: function(serverResp) {            // on ajax failure
          console.log('5xx http status code or unexpected response content');
      }
  });
